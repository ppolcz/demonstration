function paren_array(k,l,m,n)
  for i = k,m do
    for j = l,n do
      tex.sprint("("..i..","..j..")  ")
    end
    tex.print("")  -- force a line break
  end
end

function subscript_array(ind,k,l,m,n)
  for i=k,m do
    for j=l,n do
      tex.sprint(ind.."_{"..i..j.."}")
      if j<n then
        tex.sprint ( "&" )    -- cell separator (ampersand)
      else
        tex.sprint( "\\\\" )  -- end of row (two backslashes)
      end
    end
  end
end
